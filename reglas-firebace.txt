rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{uid} {
      allow read, write: if request.auth.uid == uid;
    }
    match /proyectos/{uid} {
      allow read, write: if request.auth.uid == uid;
    }
    match /proyectos-publicos/{uid} {
      allow read: if true;
      allow write: if request.auth.uid == uid;
    }
  }
}
